using System;

using System.Text;

        {

            Console.WriteLine("Car Insurance Approval");
            Console.WriteLine();
          
            //Age of applicant//
            Console.WriteLine("What is your age?");
            int age = Convert.ToInt32(Console.ReadLine());

	    bool qualiFied

            //Qualification for Insurance//
